// This is our controller
function getValues() {
  //get the user's input
  //decide what to do with it
  let userInput = document.getElementById('message').value;

  let noPunctuationOrSpacing = userInput.toLowerCase().replace(
    /[.,/#!$%^&*;:{}=-_`~() ]/g,
    ""
  );

  let reversedInput = reverseString(noPunctuationOrSpacing);

  displayString(reversedInput);
}

// logic
function reverseString(messages) {
  //take a string, return that string in reverse
  let userInput = document.getElementById("message").value;
  let noPunctuationOrSpacing = userInput
    .toLowerCase()
    .replace(/[.,/#!$%^&*;:{}=-_`~() ]/g, "");
  let reversedMessage = ''

  for (let index = messages.length - 1; index >= 0; index--) {
    reversedMessage = reversedMessage + messages[index];
  }

  if (noPunctuationOrSpacing == reversedMessage) {
    finalMessage = "Correct, " + userInput + " is a palindrome";
  } else
    finalMessage =
      "Sorry, but it seems that " +
      userInput +
      " is not a palendrome, try again!";

  return finalMessage
}

//view function
function displayString(reversedMessage) {
  //show the string on the page
  document.getElementById('msg').textContent = reversedMessage;
  document.getElementById('alert').classList.remove('d-none');
}

A brief overview of the code itself

How it works...

First we begin with our with our controller function. The purpose of this fuction is to pull everything from the html, then take the code that has been changed later and put it in our view function.

Next we see wihtin our "logic fuction" that we take the inital message from the html and compare it to the new message results(we will call this initial message m1). The new message results are produced by taking our original message (seperate from m1 we will call this original message m2) and flipping it around. We then compare m1 with m2 and see if they are the same. If they are then we get a message that our message is a palendrome, if not we recieve the message that it is not a palendrome. It is important to say any and all characters that are not letters/numbers are removed in the code in order to find a result even if you typed "tacocat!!!" then you would still get tacocat. This also works for capitalization.

Lastly the code is moved back into our original function, then down to our view function. Where the code is sent to our HTML so that you can see all of it on the page!
